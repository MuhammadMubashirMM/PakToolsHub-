🖼️ Image Compressor







Download Image function compressImage(){ let file=document.getElementById("imageInput").files[0]; if(!file){ alert("Please select an image"); return; } let reader=new FileReader(); reader.onload=function(e){ document.getElementById("preview").src=e.target.result; document.getElementById("preview").style.display="block"; let download=document.getElementById("downloadBtn"); download.href=e.target.result; download.download="image.jpg"; download.style.display="inline"; } reader.readAsDataURL(file); }
